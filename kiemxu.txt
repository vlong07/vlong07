def get_info(headers):
    try:
        response = requests.get('https://dvmxh.azdigi.blog/diemdanh.php', headers=headers)
        response.raise_for_status()
        soup = BeautifulSoup(response.text, 'html.parser')
        return soup.find_all('h3', class_='gradient-text mb-3')
    except requests.RequestException as e:
        print(f"Error fetching info: {e}")
        return None

def draw_game(headers):
    try:
        response = requests.post('https://dvmxh.azdigi.blog/game.php', headers=headers, data={'draw': 'Rút Ngay !'})
        response.raise_for_status()
        soup = BeautifulSoup(response.text, 'html.parser')
        result_div = soup.find('div', class_='result mt-4')
        if result_div:
            alert_div = result_div.find('div', class_='alert alert-success')
            if alert_div:
                return alert_div.text.strip()
    except requests.RequestException as e:
        print(f"Error drawing game: {e}")
    return None

def buy_draws(headers):
    try:
        response = requests.post('https://dvmxh.azdigi.blog/game.php', headers=headers, data={'buy_draws': 'Mua 55 lượt với 1k2 xu'})
        response.raise_for_status()
        return True
    except requests.RequestException as e:
        print(f"Error buying draws: {e}")
        return False
